# ---------------------------------------------------------------
# Подготовка
# установка компилятора "nasm" (можно делать из любой директории)
sudo apt install nasm

# все описанное ниже необходимо делать в той же директории 
# где и лежит сам файл с исходным кодом "filename.asm"
# имя файла "filename" приведено в качестве примера

# переход в директорию, в которой лежит файл "filename.asm"
cd <путь до директории с файлом "filename.asm">
# например:
cd /home/user/Documents/
# ---------------------------------------------------------------

# создание объектного файла "filename.o" из файла с исходным кодом "filename.asm"
nasm -f elf filename.asm

# создание исполяемого файла "filename" из объектного файла "filename.o"
ld -m elf_i386 -o filename filename.o

# добавление прав на исполнение для исполняемого файла "filename"
chmod a+x filename

# запуск исполняемого файла "filename"
./filename
